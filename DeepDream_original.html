import tensorflow as tf
from PIL import Image
import numpy as np
import urllib.request

# Define the model
model = tf.keras.applications.InceptionV3(include_top=False, weights='imagenet')

# Define a function to apply DeepDream
def deepdream(model, img, steps=200, step_size=0.01):
    for step in range(steps):
        with tf.GradientTape() as tape:
            tape.watch(img)
            preds = model(img)
            loss = tf.math.reduce_sum(preds)
        grads = tape.gradient(loss, img)
        grads /= tf.math.reduce_std(grads) + 1e-8
        img = img + grads * step_size
    return img

# Load an image
def load_image(img_path, max_size=400, shape=None):
    image = Image.open(img_path)
    if max(image.size) > max_size:
        size = max_size
    else:
        size = max(image.size)
    if shape is not None:
        size = shape
    image = image.resize((size, size))
    return np.array(image)

# load the content image
content = load_image('content.png', max_size=400)

# Convert the image to a format suitable for processing
content = tf.keras.applications.inception_v3.preprocess_input(content)
content = tf.expand_dims(content, axis=0)

# Generate DeepDream image
content = deepdream(model, content, steps=100, step_size=0.01)

# Postprocessing
content = tf.squeeze(content, axis=0)
content = content - tf.reduce_min(content)
content = content / tf.reduce_max(content)
content = np.array(content * 255, dtype=np.uint8)
content = Image.fromarray(content)

content.save("deepdream.jpg")
content.show()
